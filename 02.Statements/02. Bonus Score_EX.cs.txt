using System;

namespace ConditionalStatementsExercise
{
    class Program
    {
        static void Main(string[] args)
        {
           
            //1. Четем цяло число от конзолата
            int number = int.Parse(Console.ReadLine());
            double bonus = 0.0;
            //2. Проверяваме дали числото е <=100. Ако е, добавяме 5 бонус точки;
            if (number<=100)
            {
                bonus = 5;
            }
            //3. Проверяваме, дали числото>100. Ако е, добавяме 20% бонус точки
            else if(number>1000)
            {
                bonus = number * 0.1;
            }
            //4. Проверяваме, дали числото>1000.Ако е, добавяме 10% бонус точки
            else 
            {
                bonus = number * 0.2;
            }
            //5. Начисляваме допълнителни бонус точки
            // а) Ако числото е четно + 1 т.
            if (number%2==0)
            {
                bonus = bonus + 1;
            }
            //б) Ако числото завършва на 5 + 2 т.
            else if(number%10==5)
            {
                bonus = bonus + 2;
            }
            //6.Изпечатваме бонуса на конзолата
            Console.WriteLine(bonus);

            //7.Изпечатване общия брой точки
            Console.WriteLine(number+bonus);
        }
    }
}
