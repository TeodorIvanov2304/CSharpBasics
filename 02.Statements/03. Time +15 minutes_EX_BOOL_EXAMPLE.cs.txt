using System;

namespace ConditionalStatementsExercise
{
    class Program
    {
        static void Main(string[] args)
        {
            // Да се напише програма, която чете час и минути от 24-часово денонощие,
            // въведени от потребителя и изчислява колко ще е часът след 15 минути.
            // Резултатът да се отпечата във формат часове:минути. Часовете винаги са между 0 и 23,
            // а минутите винаги са между 0 и 59. Часовете се изписват с една или две цифри.
            // Минутите се изписват винаги с по две цифри, с водеща нула, когато е необходимо. 

            //1. Четем от конзолата: час и минути
            int hours = int.Parse(Console.ReadLine());    //1
            int minutes = int.Parse(Console.ReadLine());  //20

            //2. Преобразуваме час и мин. само в минути
            minutes = minutes + hours * 60;              // 20+1*60 = 20+60=80
            // minutes +=hours * 60
            //3. Към мин. добавяме 15
            minutes = minutes + 15;                      // 80+15=95
            // minutes+=15;

            //4. Преобразуваме времето в мин и часове.
            hours = minutes / 60;                       // 95/60=1
            minutes = minutes % 60;                     // 95%60=35

            //5. Проверяваме дали сме отишли в следващия ден (часовете==24)
            bool newDay = hours == 24;  //false

            if (newDay)
            {
                Console.WriteLine($"0:{minutes:D2}");
            }
            else
            {
                Console.WriteLine($"{hours}:{minutes:D2}");
            }
            //6. Отпечатваме на конзолата
            // * мин. се отпечатват с водеща 0 ако са <10
            
            
        }
    }
}
