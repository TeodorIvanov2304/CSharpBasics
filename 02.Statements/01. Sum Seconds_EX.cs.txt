using System;

namespace ConditionalStatementsExercise
{
    class Program
    {
        static void Main(string[] args)
        {
            //1. Четем от конзолата времето на всеки един от състезателите
            int first = int.Parse(Console.ReadLine());
            int second = int.Parse(Console.ReadLine());
            int third = int.Parse(Console.ReadLine());

            //2. Сумираме времето в секунди
            int totalTime = first + second + third; //110

            //3. Преобразуваме времето в мин. и сек.
            int min = totalTime / 60; // 110/60 = 1
            int sec = totalTime % 60; // 110%60 = 50

            //4. Отпечатваме резултата на конзолата
            if (sec<10)
            {
                Console.WriteLine($"{min}:0{sec}");
            }
            else {
                Console.WriteLine($"{min}:{sec}");
                 }

            //=> ако секундите са по-малко от 10 => отпечатваме с водеща 0
            //=> ако не са, отпечатваме стандартно

        }
    }
}
