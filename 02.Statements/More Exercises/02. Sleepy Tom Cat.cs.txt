using System;

namespace _02._Sleepy_Tom_Cat
{
    internal class Program
    {
        static void Main(string[] args)
        {   
            //1. Четем почивните дни от конзолата
            int daysOff = int.Parse(Console.ReadLine());

            //2. Създаваме променлива, за нормата минути на Том
            int playNorm = 30000;
            //3. Намираме работните дни за годината
            int workDays = 365 - daysOff;

            //4. Намираме общото време за игра в минути
            int timeForPlay = (daysOff * 127) + (workDays * 63);

            //5. Проверявам, дали времето на Том > или < от нормата
            // a Дали  е по-голямо и отпечатваме на конзолата :
            if(timeForPlay>playNorm)
            {
                int difference = timeForPlay - playNorm;
                int hours = difference / 60;
                int minutes = difference % 60;
                Console.WriteLine("Tom will run away");
                Console.WriteLine($"{hours} hours and {minutes} minutes more for play");
            }
            // б Дали е по-малко и отпечатваме на конзолата
            else 
            {
                int difference = playNorm - timeForPlay;
                int hours = difference / 60;
                int minutes = difference % 60;
                Console.WriteLine("Tom sleeps well");
                Console.WriteLine($"{hours} hours and {minutes} minutes less for play");
              
            }


            

        }
    }
}
