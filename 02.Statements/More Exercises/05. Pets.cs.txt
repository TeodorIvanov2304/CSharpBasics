using System;
namespace _05._Pets
{
    internal class Program
    {
        static void Main(string[] args)
        {
            //Марина обича да пътува. Тя има 3 домашни любимеца (куче, котка и костенурка).
            //Когато заминава на пътешествие трябва да съобрази колко храна да им остави,
            //за да не останат гладни. Напишете програма, която пресмята колко килограма
            //храна ще изядат всички за времето, в което Марина отсъства и дали оставената
            //храна от нея ще им е достатъчна. Всяко животно изяжда определено количество храна на ден
            // 1. Четем от конзолата 5 реда 
            // Първи ред – брой дни – int[1…5000]
            // Втори ред – оставена храна в килограми – int [0…100000]
            // Трети ред – храна на ден за кучето в килограми – double [0.00…100.00]
            // Четвърти ред – храна на ден за котката в килограми– double [0.00…100.00]
            // Пети ред – храна на ден за костенурката в грамове – double л[0.00…10000.00]
            
            int days = int.Parse(Console.ReadLine());
            int foodKilos = int.Parse(Console.ReadLine());
            double dogFoodKilosPerDay = double.Parse(Console.ReadLine());
            double catFoodKilosPerDay = double.Parse(Console.ReadLine());
            double turtleFoodGramsPerDay = double.Parse(Console.ReadLine());

            double petFoodKilos = (dogFoodKilosPerDay * days) + (catFoodKilosPerDay * days) + (turtleFoodGramsPerDay * days / 1000);

            if (foodKilos>=petFoodKilos)
            {
                Console.WriteLine($"{Math.Floor(foodKilos - petFoodKilos)} kilos of food left.");
            }
            else
            {
                Console.WriteLine($"{Math.Ceiling(petFoodKilos-foodKilos)} more kilos of food are needed.");
            }
        }
    }
}
