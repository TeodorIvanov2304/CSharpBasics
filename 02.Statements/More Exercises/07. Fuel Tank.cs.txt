using System;

namespace _07._Fuel_Tank
{
    internal class Program
    {
        static void Main(string[] args)
        {

            //Напишете програма, която познава дали резервоара на едно превозно
            //средство има нужда от презареждане на горивото или не. От конзолата
            //се четат два реда – текст и реално число, на първия ред се чете типа на горивото
            //– текст с възможности: "Diesel", "Gasoline" или "Gas", а на втория литрите гориво,
            //които има в резервоара. Ако литрите гориво са повече или равни на 25, на конзолата
            //да се отпечата "You have enough {вида на горивото}.", ако са по-малко от 25,
            //да се отпечата "Fill your tank with {вида на горивото}!". В случай, че бъде въведено гориво,
            //различно от посоченото, да се отпечата "Invalid fuel!".

            //1. Четем от конзолата типът гориво и литрите
            string fuelType = Console.ReadLine(); // .То
            double fuelLiters = double.Parse(Console.ReadLine());

            if(fuelType=="Diesel"&&fuelLiters>=25.0)
            {
                fuelType = "diesel";
                Console.WriteLine($"You have enough {fuelType}.");
            }
            else if(fuelType=="Diesel"&&fuelLiters<25.0)
            {
                fuelType = "diesel";
                Console.WriteLine($"Fill your tank with {fuelType}!");
            }
            else if(fuelType=="Gasoline"&&fuelLiters>=25.0)
            {
                fuelType = "gasoline";
                Console.WriteLine($"You have enough {fuelType}.");
            }
            else if(fuelType == "Gasoline" && fuelLiters < 25.0)
            {
                fuelType = "gasoline";
                Console.WriteLine($"Fill your tank with {fuelType}!");
            }
            else if(fuelType=="Gas"&&fuelLiters>=25.0)
            {   
                fuelType = "gas";
                Console.WriteLine($"You have enough {fuelType}.");
            }
            else if (fuelType == "Gas" && fuelLiters < 25.0)
            {
                fuelType = "gas";
                Console.WriteLine($"Fill your tank with {fuelType}!");
            }
            else
            {
                Console.WriteLine("Invalid fuel!");
            }
        }
    }
}
