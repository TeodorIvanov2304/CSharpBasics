using System;

namespace _06._Flower_Shop
{
    internal class Program
    {
        static void Main(string[] args)
        {

            //Мария иска да купи подарък на сина си. Тя работи в магазин за цветя.
            //В магазина идва поръчка за цветя. Напишете програма, която пресмята
            //сумата от поръчката и дали печалбата е достатъчна за подаръка.
            //Цветята имат следните цени:
            // Магнолии – 3.25 лева
            // Зюмбюли – 4 лева
            // Рози – 3.50 лева
            // Кактуси – 8 лева
            // От общата сума, Мария трябва да плати 5 % данъци.

            //1. Четем от конзолата 5 реда:
            //Брой магнолии – цяло число в интервала [0 … 50]
            //  Брой зюмбюли – int [0 … 50]
            //  Брой рози – int [0 … 50]
            //  Брой кактуси – int  [0 … 50]
            //  Цена на подаръка – int [0.00 … 500.00]
            //  
            int magnolias = int.Parse(Console.ReadLine());
            int hyacinths = int.Parse(Console.ReadLine());
            int roses = int.Parse(Console.ReadLine());
            int cacti = int.Parse(Console.ReadLine());
            double present = double.Parse(Console.ReadLine());

            //2. Пресмятаме цената на поръчката
            double orderPrice = (magnolias * 3.25) + (hyacinths * 4.0) + (roses * 3.50) + (cacti * 8.0);
            //3. Изваждаме данъка
            double finalPrice = orderPrice - (orderPrice * 0.05);

            //4. Сравняваме дали цената на подаръка е < или >
            // от цената на поръчката и изпечатваме на конзолата
            if(finalPrice >= present)
            {
                Console.WriteLine($"She is left with {Math.Floor(finalPrice - present)} leva.");
            }
            else
            {
                Console.WriteLine($"She will have to borrow {Math.Ceiling(present- finalPrice)} leva.");
            }
        }
    }
}
