using System;
using System.Threading;

namespace CelsiusToFahrenheit
{
    class Program
    {
        static void Main(string[] args)
        {
            // Напишете програма, която да пресмята колко литра боя е нужна за боядисването на къщa.
            // Като за стените се използва зелена боя, а за покрива – червена.
            // Разхода на зелената боя е 1 литър за 3.4 м2, а на червената – 1 литър за 4.3 м2.
            // Стените имат следните размери:
            // Предната и задната стена са квадрати със страна „x“
            // o на предната стена има правоъгълна врата с широчина 1.2м и височина 2м
            // • Страничните стени са правоъгълници със страни „x“ и „y“
            // o и на двете странични стени има по един квадратен прозорец със страна 1.5м
            // Покривът има следните размери:
            // •	Два правоъгълника със страни „x“ и „y“
            // •	Два равностранни триъгълника със страна „x“ и височина „h“
            // Трябва да пресметнете площта на всички страни и площта на покрива,
            // за да намерите колко литра от всяка боя ще са нужни.
            // От конзолата се четат 3 реда:
            //  1.x – височината на къщата – реално число в интервала[2...100]
            //  2.y – дължината на страничната стена – реално число в интервала[2...100]
            //  3.h – височината на триъгълната стена на прокрива – реално число в интервала[2...100]
            //

            //1. Четем от конозлата 	
            double houseHeight = double.Parse(Console.ReadLine());
            double houseLength = double.Parse(Console.ReadLine());
            double heightTriangleWall = double.Parse(Console.ReadLine());

            //2. Задаваме променливи за площта на вратите и прозороците
            double frontDoor = 1.2 * 2;
            double window = 1.5 * 1.5;

            //3. Пресмятаме лицата на предната и задната страна 
            double frontSide = (houseHeight * houseHeight) - frontDoor;
            double backSide = (houseHeight * houseHeight);
            double frontBack = frontSide + backSide;

            //4. Пресмятаме лицата на страничните стени
            double sideWall = (houseLength * houseHeight)-window;
            double roofSide = (houseHeight * houseLength)*2;
            double roofFrontBack = (heightTriangleWall * houseHeight / 2) * 2;
            
            //5. Пресмятаме цялата площ на покрива и къщата
            double roofArea = roofFrontBack + roofSide;
            double houseArea = frontBack + (sideWall*2);

            //6. Пресмятаме необходимата боя за къщата и покрива
            double greenPaint = houseArea/3.4;
            double redPaint = roofArea/4.3;

            //7. Отпечатваме резултата на конзолата
            Console.WriteLine($"{greenPaint:F2}");
            Console.WriteLine($"{redPaint:F2}");

        }
    }
}
