using System;

namespace _02._Exam_Preparation
{
    internal class Program
    {
        static void Main(string[] args)
        {

            // 1. Четем брой незадоволително оценки, които може да получи
            int failedTimes = int.Parse(Console.ReadLine());
            // 2. Създаваме променливи за:
            string input = Console.ReadLine();
            //-> intput (четем го)
            string lastExersiece = string.Empty;
            //-> последното име на задачата (празен стринг)
            int evoluationSum = 0;
            //-> променлива която ще пази сумата на всички оценки
            int evoluationCount = 0;
            //-> променлива за броя на всички задачи
            int filledCount = 0;
            //-> за броя на незадоволителните оценки, които е получил
            int evoluation;
            //-> променлива която ще пази текуащ оценка

            // 3. Създаваме цикъл, който да се изпълнява докато input!="Enough"
            while (input!="Enough")
            {
                //-> присвояваме като име на последна задача input
                lastExersiece = input;
                //-> четем текуща оценка
                evoluation = int.Parse(Console.ReadLine());

                if (evoluation<=4)
                {
                    // - увеличаваме брояча на бр. незадоволителни оценки с 1
                    filledCount++;

                    //-> проверяваме дали е достигнат бр. незадоволителни оценки
                    //-> ако е отпечатваме "You need a break, {брой незадоволителни оценки} poor grades."
                    //-> break;
                    if (filledCount==failedTimes)
                    {
                        Console.WriteLine($"You need a break, {failedTimes} poor grades.");
                        break;
                    }
                }
                evoluationSum += evoluation;
                evoluationCount++;
                input = Console.ReadLine();
            }
            if(input=="Enough")
            {
                Console.WriteLine($"Average score: {(double)evoluationSum/evoluationCount:F2}");
                Console.WriteLine($"Number of problems: {evoluationCount}");
                Console.WriteLine($"Last problem: {lastExersiece}");
            }    
            
            
            

            
            
            

            //-> добавяме текущата оценка към бр. на всички оценки
            //-> увеличаваме брояча на всички оценки с 1

            //-> ако е отпечатваме :
            // "Average score: {средна оценка}"
            // "Number of problems: {броя на всички задачи}"
            // "Last problem: {името на последната задача}"
