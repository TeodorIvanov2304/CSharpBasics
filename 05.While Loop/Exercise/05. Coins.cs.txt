using System;

namespace _05._Coins
{
    internal class Program
    {
        static void Main(string[] args)
        {
            // 2 лв. = 200 стотинки;
            // 1 лв. = 100 стотинки
            // 50 лв. = 50 стотинки
            // 20 лв. = 20 стотинки
            // 10 лв. = 10 стотинки
            // 0,05 лв. = 5 стотинки
            // 0.02 лв. = 2 стотинки
            // 0.01 лв. = 1 стотинка

            //1. Четем ресто, което трябва да върнем от конзолата и го превръщаме в стотинки(*100)
            double changeToReturn = double.Parse(Console.ReadLine())*100;

            // Създаваме брояч на монети
            int count = 0;
            //2. Създаваме цикъл, който да се изпълнява докато имаме ресто за връщане
            while (changeToReturn>0) 
            {
                //-> проверяваме коя е най-голямата монета, която можем да върнем
                //-> връщаме
                //-> увеличаваме броя на монетите, които сме върнали с 1
                if (changeToReturn>=200)
                {
                    changeToReturn = changeToReturn - 200;
                }
                else if(changeToReturn>=100)
                {
                    changeToReturn = changeToReturn - 100;
                }
                else if(changeToReturn>=50)
                {
                    changeToReturn = changeToReturn - 50;
                }
                else if(changeToReturn>=20)
                {
                    changeToReturn = changeToReturn - 20;
                }
                else if (changeToReturn>=10)
                {
                    changeToReturn = changeToReturn - 10;
                }
                else if( changeToReturn>=5)
                {
                    changeToReturn = changeToReturn - 5;
                }
                else if(changeToReturn >= 2)
                {
                    changeToReturn = changeToReturn - 2;
                }
                else if(changeToReturn>=1)
                {
                    changeToReturn = changeToReturn - 1;
                }
                else
                {
                                        // това се правим защото double changeToReturn не връща
                    break;              // много точни стойности
                }

                count++;
            }


            //3. Отпечатваме с колко монети сме върнали рестото
            Console.WriteLine(count);
        }
    }
}
