using System;

namespace _04._Walking
{
    internal class Program
    {
        static void Main(string[] args)
        {
            //1. Създаваме променливи:  
            //-> брой стъпки, които е направила
            int steps = 0;
            //-> вход от конзолата (string)
            string input;
            //2. Създаваме цикъл, който да се изпълнява докато 
            //-> Тя достигне 10 000 крачки
            //-> тя се прибира, и се пише командата Going home + още един последен инпут
            while (steps<10000) 
            {
                input = Console.ReadLine();
                // Проверяваме дали входът е "Going home"
                if (input == "Going home")
                {
                    //-> четем стъпките, които прави до вкъщи и ги добавяме към общия брой стъпки
                    steps = steps + int.Parse(Console.ReadLine());
                    break;
                }
                // Към общият брой стъпки да добавим стъпките, които са вход от конзолата
                steps = steps + int.Parse(input);
            }

            if(steps>=10000)
            {
                Console.WriteLine("Goal reached! Good job!");
                Console.WriteLine($"{Math.Abs(steps-10000)} steps over the goal!");

            }
            else
            {
                Console.WriteLine($"{10000 - steps} more steps to reach goal.");
            }


            //-> проверяваме дали стъпките са < от 10 000:
            //-> ако са по-малко: "{разликата между стъпките} more steps to reach goal."
            //->спираме цикъла break;



            //2. Проверяваме, дали е успяла да постигне целта си
            //-> ако е : "Goal reached! Good job!"  "{разликата между стъпките} steps over the goal!"

        }
    }
}
