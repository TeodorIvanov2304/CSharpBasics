using System;

namespace _04._Even_Powers_of_2
{
    internal class Program
    {
        static void Main(string[] args)
        {
            
            int n = int.Parse(Console.ReadLine());
            int num = 1;   // 2 na 0 = 1
            for (int i = 0; i <= n; i =i+2) // i+=2 за да открием четние числа
            {
                Console.WriteLine(num);
                num = num * 2 * 2; // num започва 1, така че се умножава по 2, и после още по 2 за да се получат степените на 2
            }
        }
    }
}
