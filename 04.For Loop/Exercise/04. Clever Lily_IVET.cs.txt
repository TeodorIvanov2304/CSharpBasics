using System;

namespace _04._Clever_Lily_IVET
{
    internal class Program
    {
        static void Main(string[] args)
        {
            // 1. Четем входа от конзолата
            int age = int.Parse(Console.ReadLine());
            double washingMashine = double.Parse(Console.ReadLine());
            int toyPrice = int.Parse(Console.ReadLine());

            int money = 0;
            // 2. Създаваме цикъл, който да се изпълнява за всеки РД
            for(int birthday = 1; birthday < age;birthday++)
            {
                // проверяваме дали текущия РД е четен/нечетен
                //-> ако е четен:
                if (birthday%2==0)
                {
                    //-> към спестяванията до момента добавяме += възрастта * 5 - 1(за брат й)
                    money += birthday * 5 -1;
                }
                //-> ако е нечетен
                else
                {
                    //-> към спестяванията до момента += цена на играчката
                    money += toyPrice;
                }
            }
            // 3. Проверяваме, дали парите са достатъчни да се купи пералня
            if(money>=washingMashine)
            {
                // ако са 
                Console.WriteLine($"Yes {money-washingMashine:F2}");
            }
            else
            {
                // ако не са
                Console.WriteLine($"No {washingMashine-money:F2}");
            }

            
             
            
            
            
        }
    }
}
