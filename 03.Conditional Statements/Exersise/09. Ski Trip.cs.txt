using System;

namespace _09._Ski_Trip
{
    internal class Program
    {
        static void Main(string[] args)
        {
            int days = int.Parse(Console.ReadLine()); // дни престой
            string roomType = Console.ReadLine(); // room for one person , apartment, president apartment
            string feedback = Console.ReadLine(); // positive / negative

            // крайната цена за престоя = бр. нощувки * цена за 1 нощувка
            int nights = days - 1; // бр. нощувки
            double pricePerNight = 0; // цена за нощувка -> зависи от типа на стаята 
            bool isPositive = feedback == "positive";
            // roomType -> серия от проверки за точни стойности
            //  "room for one person" – 18.00 лв за нощувка
            //	"apartment" – 25.00 лв за нощувка
            //	"president apartment" – 35.00 лв за нощувка
            switch(roomType)
            {
                case "room for one person":
                    pricePerNight = 18.0;
                    break;
                case "apartment":
                    pricePerNight = 25.00;
                    break;
                case "president apartment":
                    pricePerNight = 35.00;
                    break;
            }
            double totalPrice = pricePerNight * nights; // крайна цена

            //2 отстъпки
            if(roomType == "apartment")
            {
                // проверка за дни
                if(days<10)
                {
                    // - 30%
                    totalPrice = totalPrice - 0.30 * totalPrice;

                }
                else if(days>=10&&days<15)
                {   
                    // -35%
                    totalPrice = totalPrice - 0.35 * totalPrice;
                }
                else if(days>15)
                {   
                    // -50%
                    totalPrice = totalPrice - 0.50 * totalPrice;
                }
            }
            if(roomType == "president apartment")
            {
                if (days < 10)
                {
                    // - 10%
                    totalPrice = totalPrice - 0.10 * totalPrice;

                }
                else if (days >= 10 && days < 15)
                {
                    // -15%
                    totalPrice = totalPrice - 0.15 * totalPrice;
                }
                else if (days > 15)
                {
                    // -20%
                    totalPrice = totalPrice - 0.20 * totalPrice;
                }
            }
            if(isPositive)
            {   
                // + 25% към цената
                totalPrice = totalPrice + 0.25 * totalPrice;
                Console.WriteLine($"{totalPrice:F2}");
            }
            else
            {   
                // - 15% към цената
                totalPrice = totalPrice - 0.10 * totalPrice;
                Console.WriteLine($"{totalPrice:F2}");
            }

        }
    }
}
